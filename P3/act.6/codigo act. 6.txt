def path_finder(area):
    grid = [[int(c) for c in line] for line in area.splitlines()]
    n = len(grid)
    
    # Matriz para guardar el costo mínimo para llegar a cada celda (empezamos con infinito)
    costs = [[float('inf')] * n for _ in range(n)]
    costs[0][0] = 0
    
    queue = [(0, 0)]
    
    for r, c in queue:
        # Revisar los 4 vecinos (Arriba, Abajo, Derecha, Izquierda)
        for dr, dc in [(-1,0), (1,0), (0,1), (0,-1)]:
            nr, nc = r + dr, c + dc
            
            # Si el vecino está dentro de la matriz
            if 0 <= nr < n and 0 <= nc < n:
                # Costo actual + la diferencia de altura hacia el vecino
                new_cost = costs[r][c] + abs(grid[r][c] - grid[nr][nc])
                
                # Si encontramos un camino más corto hacia ese vecino, lo actualizamos
                if new_cost < costs[nr][nc]:
                    costs[nr][nc] = new_cost
                    queue.append((nr, nc)) # Lo agregamos a la cola para seguir explorando desde ahí
                    
    return costs[n-1][n-1]
