import heapq
from collections import Counter

def frequencies(s):
    return list(Counter(s).items()) if s else []

def build_tree(freqs):
    if len(freqs) <= 1: return None
    # Estructura en el heap: [frecuencia, ID_unico, [caracter, izq, der]]
    heap = [[f, i, [c, None, None]] for i, (c, f) in enumerate(freqs)]
    heapq.heapify(heap)
    
    unique_id = len(freqs)
    while len(heap) > 1:
        f1, _, left = heapq.heappop(heap)
        f2, _, right = heapq.heappop(heap)
        heapq.heappush(heap, [f1 + f2, unique_id, [None, left, right]])
        unique_id += 1
        
    return heap[0][2]

def get_codes(tree, path=""):
    if not tree: return {}
    if tree[0] is not None: return {tree[0]: path}
    return {**get_codes(tree[1], path + "0"), **get_codes(tree[2], path + "1")}

def encode(freqs, s):
    # Corrección: Si freqs no es válido, se rechaza inmediatamente devolviendo None
    if len(freqs) <= 1: return None
    if not s: return ""
    
    tree = build_tree(freqs)
    codes = get_codes(tree)
    return "".join(codes[char] for char in s)

def decode(freqs, bits):
    # Corrección: Si freqs no es válido, se rechaza inmediatamente devolviendo None
    if len(freqs) <= 1: return None
    if not bits: return ""
    
    tree = build_tree(freqs)
    res, curr = [], tree
    for bit in bits:
        curr = curr[1] if bit == "0" else curr[2]
        if curr[0] is not None:
            res.append(curr[0])
            curr = tree
    return "".join(res)
